# Use Node.js LTS base image
FROM node:20

# Set working directory
WORKDIR /app

# Copy package.json and package-lock.json
COPY package*.json ./

# Install dependencies
RUN npm install

# Copy all files
COPY . .

# Make a directory for auth info if not exists
RUN mkdir -p auth_info

# Expose port (if needed for panel integration)
EXPOSE 3000

# Suggest a robot AI image for bot profile (download & place manually)
# Example: https://cdn.pixabay.com/photo/2017/01/31/13/14/robot-2027195_1280.png
# Place this image as: assets/jgzbot-profile.png

# Start the bot
CMD ["npm", "start"]
